mon code JS : 

let currentQuestionIndex = 0;  // Index pour suivre la question actuelle

// Fonction pour charger et afficher les questions
function loadQuizData() {
    fetch('QuizFacile.php')
        .then(response => response.json())
        .then(data => {
            if (data.error) {
                alert(data.error);
                return;
            }

            // Références aux éléments du DOM
            let questionContainer = document.getElementById('IntituleQuestion');
            let bouton1 = document.getElementById('Bouton1');
            let bouton2 = document.getElementById('Bouton2');
            let bouton3 = document.getElementById('Bouton3');

            // Charger la question actuelle
            let currentQuestion = data[currentQuestionIndex];

            // Afficher la question
            questionContainer.innerHTML = currentQuestion.question;

            // Afficher les réponses
            bouton1.innerHTML = currentQuestion.choices[0].text;
            bouton1.value = currentQuestion.choices[0].id;

            bouton2.innerHTML = currentQuestion.choices[1].text;
            bouton2.value = currentQuestion.choices[1].id;

            bouton3.innerHTML = currentQuestion.choices[2].text;
            bouton3.value = currentQuestion.choices[2].id;
        });
}

// Fonction pour passer à la question suivante
function nextQuestion() {
    currentQuestionIndex++;
    loadQuizData();
}

// Charger la première question lorsque le bouton "Lancer Quiz" est cliqué
document.getElementById('startButton').addEventListener('click', function() {
    loadQuizData();

    // Désactiver le bouton de démarrage une fois le quiz commencé
    this.disabled = true;
    
    // Ajouter un bouton "Suivant" pour avancer à la question suivante
    let nextButton = document.createElement('button');
    nextButton.textContent = "Suivant";
    nextButton.addEventListener('click', nextQuestion);
    document.getElementById('ContainerBouton').appendChild(nextButton);
});

La structure de mes bases : 


CREATE TABLE Quiz (
    Id_Quiz INT AUTO_INCREMENT PRIMARY KEY,
    Type_Quiz ENUM('QCM', 'Vrai/Faux', 'Réponse Libre') NOT NULL,
    Niveau ENUM('Facile', 'Moyen', 'Difficile') NOT NULL,
    Date_Creation TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);


CREATE TABLE Questions (
    Id_Question INT AUTO_INCREMENT PRIMARY KEY,
    Question_Text TEXT NOT NULL,
    Id_Quiz INT,
    FOREIGN KEY (Id_Quiz) REFERENCES Quiz(Id_Quiz)
);


CREATE TABLE Choix (
    Id_Choix INT AUTO_INCREMENT PRIMARY KEY,
    Id_Question INT,
    Choix_Text VARCHAR(255) NOT NULL,
    FOREIGN KEY (Id_Question) REFERENCES Questions(Id_Question)
);


CREATE TABLE Reponse (
    Id_Reponse INT AUTO_INCREMENT PRIMARY KEY,
    Id_User INT,
    Id_Question INT,
    Reponse_Valeur TEXT,  -- Peut être un texte libre, ou un choix
    Type_Reponse ENUM('Choix', 'Vrai/Faux', 'Libre') NOT NULL,
    FOREIGN KEY (Id_User) REFERENCES Utilisateurs(Id_User),
    FOREIGN KEY (Id_Question) REFERENCES Questions(Id_Question)
);

CREATE TABLE ReponseCorrecte (
    Id_ReponseCorrecte INT AUTO_INCREMENT PRIMARY KEY,
    Id_Question INT,
    Reponse_Correcte TEXT,
    Type_Reponse TEXT,
    FOREIGN KEY (Id_Question) REFERENCES Questions(Id_Question)
);
